
        // ============ 模拟模式 ============
        let simulationEnabled = false;

        async function toggleSimulation() {
            try {
                const resp = await fetch('/api/simulation/toggle', { method: 'POST' });
                const data = await resp.json();
                simulationEnabled = data.enabled;
                updateSimulationUI();
                if (simulationEnabled) {
                    await loadSimScenes();
                }
                refreshData();
            } catch (e) {
                console.error('模拟模式切换失败:', e);
            }
        }

        async function loadSimScenes() {
            try {
                const resp = await fetch('/api/simulation/status');
                const data = await resp.json();
                const scenes = data.scenes || [];
                const container = document.getElementById('sim-scene-buttons');
                if (container) {
                    container.innerHTML = scenes.map(s =>
                        '<button onclick="applySimScene(\'' + s.name + '\')" style="background:#f0f7f4;color:#2d5a40;border:1px solid #a4e4ce;padding:8px 14px;border-radius:18px;cursor:pointer;font-size:12px;transition:all 0.3s;" ' +
                        (s.name === data.current_scene ? 'class="sim-active"' : '') +
                        '>' + s.name + '<br><span style="font-size:10px;color:#999;">' + s.description + '</span></button>'
                    ).join('');
                }
                document.getElementById('sim-scene-name').textContent = data.current_scene || '自定义';
            } catch (e) {}
        }

        async function applySimScene(sceneName) {
            try {
                await fetch('/api/simulation/scene', {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({scene: sceneName})
                });
                refreshData();
                loadSimScenes();
            } catch (e) {
                console.error('场景切换失败:', e);
            }
        }

        async function applyCustomSimulation() {
            const data = {
                power: parseFloat(document.getElementById('sim-power').value) || 150,
                voltage: parseFloat(document.getElementById('sim-voltage').value) || 220,
                current: parseFloat(document.getElementById('sim-current').value) || 680,
                temperature: parseFloat(document.getElementById('sim-temp').value) || 25,
                humidity: parseFloat(document.getElementById('sim-humidity').value) || 55,
                led: document.getElementById('sim-led').value,
                buzzer: document.getElementById('sim-buzzer').value === 'true',
                switch_on: document.getElementById('sim-switch').value === 'true'
            };
            try {
                await fetch('/api/simulation/custom', {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify(data)
                });
                refreshData();
            } catch (e) {
                console.error('自定义模拟失败:', e);
            }
        }

        async function checkSimulationStatus() {
            try {
                const resp = await fetch('/api/simulation/status');
                const data = await resp.json();
                simulationEnabled = data.enabled;
                updateSimulationUI();
                if (simulationEnabled) loadSimScenes();
            } catch (e) {}
        }

        function updateSimulationUI() {
            const bar = document.getElementById('simulation-bar');
            const toggleBtn = document.getElementById('sim-toggle-btn');
            const scenes = document.getElementById('sim-scenes');
            
            if (bar) bar.style.display = simulationEnabled ? 'block' : 'none';
            if (toggleBtn) {
                toggleBtn.textContent = simulationEnabled ? '关闭模拟模式' : '开启模拟模式';
                toggleBtn.style.background = simulationEnabled ? '#ff6b6b' : '#7ba890';
            }
            if (scenes) scenes.style.display = simulationEnabled ? 'block' : 'none';
        }

        function updateLEDBuzzer(data) {
            const ledColor = data.led || data.led_color || 'green';
            const buzzerOn = data.buzzer === true || data.buzzer === 'true';
            
            const indicator = document.getElementById('led-indicator');
            const ledText = document.getElementById('led-text');
            const buzzerIcon = document.getElementById('buzzer-icon');
            const buzzerText = document.getElementById('buzzer-text');
            const row = document.getElementById('led-buzzer-row');
            
            if (!row) return;
            row.style.display = 'flex';
            
            const colorMap = {
                'green': { color: '#4caf50', text: '正常', shadow: 'rgba(76,175,80,0.6)' },
                'yellow': { color: '#ffc107', text: '预警', shadow: 'rgba(255,193,7,0.6)' },
                'red': { color: '#f44336', text: '告警', shadow: 'rgba(244,67,54,0.6)' },
                'off': { color: '#666', text: '熄灭', shadow: 'rgba(100,100,100,0.3)' }
            };
            const cm = colorMap[ledColor] || colorMap['off'];
            if (indicator) {
                indicator.style.background = cm.color;
                indicator.style.boxShadow = '0 0 12px ' + cm.shadow;
            }
            if (ledText) {
                ledText.textContent = cm.text;
                ledText.style.color = cm.color;
            }
            
            if (buzzerIcon) buzzerIcon.textContent = buzzerOn ? '🔊' : '🔇';
            if (buzzerText) {
                buzzerText.textContent = buzzerOn ? '鸣响中' : '静音';
                buzzerText.style.color = buzzerOn ? '#f44336' : '#999';
            }
        }
