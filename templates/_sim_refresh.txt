                // 更新模拟模式数据
                if (simulationEnabled && currentDevice) {
                    try {
                        const simResp = await fetch('/api/simulation/data?device_id=' + currentDevice.device_id);
                        const simData = await simResp.json();
                        if (simData.simulation) {
                            updateLEDBuzzer(simData);
                            document.getElementById('power-value').textContent = simData.power_w || '0.0';
                            document.getElementById('voltage-value').textContent = simData.voltage_v || '0.0';
                            document.getElementById('current-value').textContent = simData.current_ma || '0';
                            document.getElementById('temperature-value').textContent = simData.temperature_c || '--';
                            document.getElementById('humidity-value').textContent = simData.humidity_percent || '--';
                            if (document.getElementById('sim-scene-name')) {
                                document.getElementById('sim-scene-name').textContent = simData.scene || '自定义';
                            }
                        }
                    } catch (e) {}
                } else {
                    updateLEDBuzzer({led: 'off', buzzer: false});
                }
